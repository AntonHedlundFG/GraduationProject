65bool ACGameMode::TryAbilityUse(AController* inController, ACUnit* inUnit, const EItemSlots inSlot, ACGridTile* inTargetTile)
66{
67 if (!GameStateRef)
68 {
69 GameStateRef = GetGameState<ACGameState>();
70 if (!GameStateRef)
71 {
72 UE_LOG(LogTemp, Warning, TEXT("Wrong Game State class"));
73 return false;
74 }
75 }
76
77 if (GameStateRef->TurnOrder.IsEmpty())
78 {
79 UE_LOG(LogTemp, Warning, TEXT("There are no units in the turn order"));
80 return false;
81 }
82
83 if (GameStateRef->TurnOrder[0] != inUnit)
84 {
85 UE_LOG(LogTemp, Warning, TEXT("Unit is not first in the turn order"));
86 return false;
87 }
88
89 if (!inUnit->IsControlledBy(inController))
90 {
91 UE_LOG(LogTemp, Warning, TEXT("Unit is not owned by controller"));
92 return false;
93 }
94
95 UCItem* Item = inUnit->GetItemInSlot(inSlot);
96 if (!Item)
97 {
98 UE_LOG(LogTemp, Warning, TEXT("No item in this item slot"));
99 return false;
100 }
101
102 if (!Item->IsValidTargetTile(inUnit, inTargetTile))
103 {
104 UE_LOG(LogTemp, Warning, TEXT("Target tile is not valid for the item"));
105 return false;
106 }
107
108 UCCommand* NewCommand = Item->GenerateAbilityCommand(inController, inUnit, inTargetTile);
109 if (!NewCommand)
110 {
111 UE_LOG(LogTemp, Warning, TEXT("No ability command on this item"));
112 return false;
113 }
114
115
116 CommandList.Add(NewCommand);
117 NewCommand->ExecuteCommand(inController);
118
119 FString Log = FString("Executed command: ") + NewCommand->ToString();
120 UE_LOG(LogTemp, Warning, TEXT("%s"), *Log);
121 UCLogManager::Log(ELogCategory::LC_Gameplay,FString("Executed command: ") + NewCommand->ToString());
122 return true;
123}

125bool ACGameMode::TryUndo(AController* inController)
126{
127 if (CommandList.IsEmpty())
128 {
129 UE_LOG(LogTemp, Warning, TEXT("No commands in history"));
130 return false;
131 }
132
133 UCCommand* LastCommand = CommandList.Last();
134 if (LastCommand->GetCommandCreator() != inController)
135 {
136 UE_LOG(LogTemp, Warning, TEXT("Command was not performed by this controller"));
137 return false;
138 }
139
140 LastCommand->UndoAllConsequences();
141 LastCommand->UndoCommand();
142
143 CommandList.RemoveAtSwap(CommandList.Num() - 1);
144
145 /*FString Log = FString("Undid command: ") + LastCommand->ToString();
146 UE_LOG(LogTemp, Warning, TEXT("%s"), *Log);*/
147 UCLogManager::Log(ELogCategory::LC_Gameplay,FString("Undid command: ") + LastCommand->ToString());

151bool ACGameMode::TryEndTurn(AController* inController)
152{
153 if (!GameStateRef)
154 {
155 GameStateRef = GetGameState<ACGameState>();
156 if (!GameStateRef)
157 {
158 UE_LOG(LogTemp, Warning, TEXT("Wrong Game State class"));
159 return false;
160 }
161 }
162
163 if (GameStateRef->TurnOrder.IsEmpty())
164 {
165 UE_LOG(LogTemp, Warning, TEXT("There are no units in the turn order"));
166 return false;
167 }
168
169 ACUnit* CurrentUnit = GameStateRef->TurnOrder[0];
170 if (!CurrentUnit->IsControlledBy(inController))
171 {
172 UE_LOG(LogTemp, Warning, TEXT("Controller does not manage current unit"));
173 return false;
174 }
175
176 //Move active unit to back of the line
177 GameStateRef->TurnOrder.RemoveAt(0);
178 GameStateRef->TurnOrder.Add(CurrentUnit);
179 GameStateRef->OnRep_TurnOrder();
180
181 auto* Subsystem = GetWorld()->GetSubsystem<UCTurnTimerSubsystem>();
182 if (Subsystem)
183 {
184 Subsystem->NextTurn(CurrentUnit, GameStateRef->TurnOrder[0]);
185 }
186
187 //Transfer all commands this turn into the command history
188 for (UCCommand* Command : CommandList)
189 {
190 CommandHistory.Add(Command);
191 }
192 CommandList.Empty();
193
194 UE_LOG(LogTemp, Warning, TEXT("Turn ended"));
195 return true;
196}

209void ACGameMode::ApplyPlayerCount(const TArray<ACUnit*>& Units)
210{
211 int PlayerCount = UGameplayStatics::GetIntOption(OptionsString, NUMBER_OF_PLAYERS, 0);
212 if (!PlayerCount) return;
213 for (ACUnit* Unit : Units)
214 {
215 if (!Unit) continue;
216
217 // Might look tricky, but it's fairly simple.
218 // If player count is 2, characters with 3/4 become 1/2 instead.
219 while (Unit->ControllingPlayerIndex > PlayerCount)
220 {
221 Unit->ControllingPlayerIndex -= PlayerCount;
222 }
223 }
224}

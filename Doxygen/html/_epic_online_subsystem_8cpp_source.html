149void UEpicOnlineSubsystem::FindReconnectableGame(int ReconnectSlotIndex)
150{
151 IOnlineSubsystem* Subsystem = Online::GetSubsystem(GetWorld());
152 if (!Subsystem) return;
153 IOnlineSessionPtr Session = Subsystem->GetSessionInterface();
154 if (!Session) return;
155
156 OnReconnectSessionSearchUpdate.Broadcast(false);
157 auto PC = UGameplayStatics::GetPlayerController(this, 0);
158 auto NetID = GetUniqueNetIdOf(PC);
159 if (!NetID.IsValid())
160 {
161 UE_LOG(LogTemp, Warning, TEXT("Could not search for reconnectable games, NetID is invalid."));
162 return;
163 }
164
165 ReconnectSessionSearch = MakeShareable(new FOnlineSessionSearch());
166 ReconnectSessionSearch->MaxSearchResults = 1;
167 ReconnectSessionSearch->QuerySettings.SearchParams.Empty();
168
169 ReconnectSessionSearch->QuerySettings.Set(GetReconnectKeyName(ReconnectSlotIndex), NetID->ToString(), EOnlineComparisonOp::Equals);
170
171 FindReconnectSessionsHandle = Session->OnFindSessionsCompleteDelegates.AddUObject(this, &UEpicOnlineSubsystem::OnFindReconnectableGamesCompleted);
172 Session->FindSessions(0, ReconnectSessionSearch.ToSharedRef());
173}

175void UEpicOnlineSubsystem::OnFindReconnectableGamesCompleted(bool bWasSuccessful)
176{
177 IOnlineSubsystem* Subsystem = Online::GetSubsystem(GetWorld());
178 if (!Subsystem) return;
179 IOnlineSessionPtr Session = Subsystem->GetSessionInterface();
180 if (!Session) return;
181
182 Session->ClearOnFindSessionsCompleteDelegates(this);
183 FindReconnectSessionsHandle.Reset();
184
185 if (bWasSuccessful)
186 {
187 UE_LOG(LogTemp, Warning, TEXT("Succeeded in searching for reconnect sessions with slot %d"), NextReconnectSlot);
188 bool bFoundSession = !ReconnectSessionSearch->SearchResults.IsEmpty();
189 OnReconnectSessionSearchUpdate.Broadcast(bFoundSession);
190 if (!bFoundSession)
191 {
192 NextReconnectSlot++;
193 FindReconnectableGames(false);
194 }
195 }
196 else
197 {
198 OnReconnectSessionSearchUpdate.Broadcast(false);
199 UE_LOG(LogTemp, Warning, TEXT("Failed in searching for reconnect sessions with slot %d"), NextReconnectSlot);
200 }
201}

217void UEpicOnlineSubsystem::UpdateReconnectablePlayerList()
218{
219 IOnlineSubsystem* Subsystem = Online::GetSubsystem(GetWorld());
220 if (!Subsystem) return;
221 IOnlineSessionPtr Session = Subsystem->GetSessionInterface();
222 if (!Session) return;
223
224 FOnlineSessionSettings Settings = *Session->GetSessionSettings(SESSION_NAME);
225
226 for (int i = 0; i < MAX_RECONNECT_SLOTS; i++)
227 {
228 if (i < DisconnectedPlayers.Num())
229 Settings.Set(GetReconnectKeyName(i), DisconnectedPlayers[i], EOnlineDataAdvertisementType::ViaOnlineService);
230 else
231 Settings.Set(GetReconnectKeyName(i), FString("EMPTY"), EOnlineDataAdvertisementType::ViaOnlineService);
232 }
233 Session->UpdateSession(SESSION_NAME, Settings);
234}

444 SessionSearchResults.Sessions.Empty();

461void UEpicOnlineSubsystem::OnJoinSessionCompleted(FName SessionName, EOnJoinSessionCompleteResult::Type Result)
462{
463 if (Result == EOnJoinSessionCompleteResult::Success)
464 {
465 if (APlayerController* PlayerController = UGameplayStatics::GetPlayerController(GetWorld(), 0))
466 {
467 IOnlineSubsystem* Subsystem = Online::GetSubsystem(GetWorld());
468 if (!Subsystem) return;
469 IOnlineSessionPtr Session = Subsystem->GetSessionInterface();
470 if (!Session) return;
471 Session->ClearOnJoinSessionCompleteDelegates(this);
472 JoinSessionHandle.Reset();
473
474 FString JoinAddress;
475 Session->GetResolvedConnectString(SessionName, JoinAddress);
476 if (!JoinAddress.IsEmpty())
477 {
478 PlayerController->ClientTravel(JoinAddress, ETravelType::TRAVEL_Absolute);
479 }
480 }
481 return;
482 }
483 if (Result == EOnJoinSessionCompleteResult::SessionIsFull)
484 {
485 UE_LOG(LogTemp, Warning, TEXT("Session is full."));
486 }
487 else
488 {
489 UE_LOG(LogTemp, Warning, TEXT("Couldn't join session, unknown reason."));
490 }
491
492 //If we fail to join a session, we automatically search for new ones.
493 FindSessions();
494}

596FUniqueNetIdRepl UEpicOnlineSubsystem::GetUniqueNetIdOf(APlayerController* Player)
597{
598 FUniqueNetIdRepl UniqueNetIdRepl;
599 if (Player->IsLocalPlayerController())
600 {
601 ULocalPlayer* LocalPlayer = Player->GetLocalPlayer();
602 if (IsValid(LocalPlayer))
603 {
604 UniqueNetIdRepl = LocalPlayer->GetPreferredUniqueNetId();
605 }
606 else
607 {
608 UNetConnection* RemoteNetConnection = Cast<UNetConnection>(Player->Player);
609 if (!IsValid(RemoteNetConnection))

611 UE_LOG(LogTemp, Warning, TEXT("Leaving player has neither valid local player nor remote"));
612 return FUniqueNetIdRepl();
613 }
614 UniqueNetIdRepl = RemoteNetConnection->PlayerId;
615 }
616 }
617 else
618 {
619 UNetConnection* RemoteNetConnection = Cast<UNetConnection>(Player->Player);
620 if (!IsValid(RemoteNetConnection))
621 {
622 UE_LOG(LogTemp, Warning, TEXT("Leaving player has no valid remote connection."));
623 return FUniqueNetIdRepl();
624 }
625 UniqueNetIdRepl = RemoteNetConnection->PlayerId;
626 }
627 return UniqueNetIdRepl;

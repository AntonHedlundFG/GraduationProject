9void AOnlineGameMode::PostLogin(APlayerController* NewPlayer)
10{
11 auto Subsystem = GetGameInstance()->GetSubsystem<UEpicOnlineSubsystem>();
12
13 //Bypass online functionality, usually in PIE mode
14 if (!Subsystem || !Subsystem->IsPlayerLoggedIn())
15 {
16 Super::PostLogin(NewPlayer);
17
18 if (auto* CastState = Cast<AOnlinePlayerState>(NewPlayer->PlayerState))
19 {
20 if (!NewPlayer->IsLocalPlayerController() && CastState->PlayerIndex <= 1)
21 {
22 CastState->PlayerIndex = NextPlayerIndex;
23 NextPlayerIndex++;
24 }
25 }
26
27 return;
28 }
29
30 Subsystem->PlayerConnected(NewPlayer);
31
32 //If we have no unique net ID we are currently not using online functionality, probably PIE.
33 const FUniqueNetIdRepl UniqueNetID = Subsystem->GetUniqueNetIdOf(NewPlayer);
34
35 const FString UniqueNetIDString = UniqueNetID->ToString();
36
37 //This is where a pawn gets spawned if the player doesn't already have one
38 Super::PostLogin(NewPlayer);
39
40 // -- HANDLE INACTIVE PLAYER STATES BELOW --
41
42 // Search for inactive states belonging to the connecting players unique player ID.
43 for (AOnlinePlayerState* InactiveState : InactiveStates)
44 {
45 if (InactiveState->UniqueOwnerNetID == UniqueNetIDString)
46 {
47 InactiveStates.RemoveSingleSwap(InactiveState);
48 InactiveState->DispatchCopyProperties(NewPlayer->PlayerState);
49 InactiveState->Destroy();
50 break;
51 }
52 }
53
54 //Update the playerstate with the unique player ID, for reconnecting.
55 if (auto* CastState = Cast<AOnlinePlayerState>(NewPlayer->PlayerState))
56 {
57 CastState->UniqueOwnerNetID = UniqueNetIDString;
58
59 if (!NewPlayer->IsLocalPlayerController() && CastState->PlayerIndex <= 1)
60 {
61 CastState->PlayerIndex = NextPlayerIndex;
62 NextPlayerIndex++;
63 }
64 }
65
66}
